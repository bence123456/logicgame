List(TileBase) tiles = getTileList();

for (TileBase tile : tiles) {
    if(tile.isHelper()) {
        int expectedAdjacentBulbsNr = tile.getIntValue(1);
        List(TileBase) neighbourTiles = getHorizontalAndVerticalNeighboursForTile(tile);

        int actualAdjacentBulbsNr = 0;
        for (int i=0; i<neighbourTiles.size()-1; i++) {
            TileBase actualTile = neighbourTiles.get(i);

            Item symbolItem = tile.getItem(1);
            if (!isNull(symbolItem) && symbolItem.equals([S1])) {
                actualAdjacentBulbsNr += 1;
            }
        }

        if (expectedAdjacentBulbsNr != actualAdjacentBulbsNr) {
            return false;
        }
    } else {
        Item colorItem = tile.getItem(0);

        if (!isNull(colorItem) && colorItem.equals([#ffffff])) {
            return false;
        }
    }
}

return true;
