List(TileBase) tiles = getTiles();

for (TileBase tile : tiles) {
    if(tile.isHelper()) {
        int skylineNumber = tile.getIntValue(1);
        int visibleSkylineNumber = 1;
        List(TileBase) playableTiles = getPlayableTilesForHelperTile(tile);

        TileBase actualTile = getTile(playableTiles, 0);
        for (int i=1; i<playableTiles.siz(); i++) {
            TileBase nextTile = playableTiles.get(i);
            int nextTileIntValue = nextTile.getIntValue(1);

            if (actualTile.getIntValue(1) < nextTileIntValue) {
                visibleSkylineNumber += 1;
                actualTile = nextTile;
            }
        }

        if(skylineNumber != visibleSkylineNumber) {
                return false;
        }
    }
}

boolean valid = numberOfItemsInAllRow([1],1) &&
            numberOfItemsInAllRow([2],1) &&
            numberOfItemsInAllRow([3],1) &&
            numberOfItemsInAllRow([4],1) &&
            numberOfItemsInAllColumn([1],1) &&
            numberOfItemsInAllColumn([2],1) &&
            numberOfItemsInAllColumn([3],1) &&
            numberOfItemsInAllColumn([4],1);
return valid;
